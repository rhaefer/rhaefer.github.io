bird1$Date<-ymd(bird1$Date)
bird_table<-data.frame(bird1 %>% group_by(Common.Name, url) %>% summarise(Count=sum(Count), Location=paste(unique(State.Province), collapse=" & "),mydates=format(max(Date), '%Y/%m/%d')))
bird_table$Location<- gsub("US-WA","Washington State",bird_table$Location)
bird_table$Location<- gsub("US-OR","Oregon",bird_table$Location)
bird_table$Location<- gsub("US-MT","Montana",bird_table$Location)
bird_table$Location<- gsub("US-CA","California",bird_table$Location)
bird_table$Location<- gsub("MX-BCS","Baja, Mexico",bird_table$Location)
bird_table$Location<- gsub("CA-BC","British Columbia",bird_table$Location)
bird_table$url <- paste0("<a href=", "\"", bird_table$url, "\"","target=", "\"", "blank", "\"", ">", bird_table$Common.Name, " @ All About Birds","</a>")
bird_table<-bird_table %>% 
      rename(`Common Name`=Common.Name, `Total Observed`=Count, Locations=Location, `More Information`=url, RecentObservation=mydates) %>% 
      dplyr::select(`Common Name`, `Total Observed`, Locations, `More Information`, RecentObservation)
bird_table$RecentObservation<-as.Date(bird_table$RecentObservation, "%Y/%m/%d")
bird_table<-bird_table[order(bird_table$RecentObservation, decreasing = TRUE),]
bird_table$RecentObservation<-format(bird_table$RecentObservation, "%B %d, %Y")
bird_table<-bird_table %>% rename (`Recent Observation`=RecentObservation)

DT::datatable(bird_table, rownames=FALSE, escape=FALSE, options=list(pageLength = 200, dom = 'ft'))
